\sctn{Program Control}
\fig{\centering\img{1}{prog}}{Program control buttons}{progctrl}
\ssctn{Run}
Start execution of the source if it had been previously stopped or paused.
\itm{
  \item Main window \rarr Play button
  \item Shortcut: F4
  \item \itl{Actions \rarr Run}\\\\
}
\ssctn{Pause}
Pause the execution of the source.
\itm{
  \item Main window \rarr Pause button
  \item Shortcut: F5
  \item \itl{Actions \rarr Pause}\\\\
}
\ssctn{Stop}
Stop the execution of the source and reset back to the start.
\itm{
  \item Main window \rarr Stop button
  \item Shortcut: F6
  \item \itl{Actions \rarr Stop}
}
